  /**
   * The code for the payment method.

  code: PaymentMethod
 
  /**
   * The pay code for the payment.
   */
  pay_code?: string
 
  /**
   * The amount of the payment.
   */
  amount?: number
 
  /**
   * Flag indicating whether HTML is allowed.

  allow_html?: boolean
}

}
